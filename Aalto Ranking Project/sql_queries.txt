SELECT * FROM ranking_project LIMIT 20

SELECT institution, country, quality_of_education, alumni_employment, quality_of_faculty, score
FROM ranking_project
WHERE year = 2015
ORDER BY score DESC
LIMIT 10;

SELECT institution, country, quality_of_education, alumni_employment, quality_of_faculty, score
FROM ranking_project
WHERE year = 2015
ORDER BY quality_of_education ASC
LIMIT 10;


SELECT country, COUNT(*) AS number_of_universities, AVG(score) AS average_score
FROM ranking_project
WHERE year = 2015
GROUP BY country
ORDER BY average_score DESC
LIMIT 10;

SELECT country, AVG(score) AS average_score, COUNT(*) AS number_of_universities
FROM ranking_project
WHERE year = 2015
GROUP BY country
ORDER BY average_score DESC;
